Haz una funcion recursiva para calcular el termino n de la seria de Fibonacci. El n-terme
de la serie de Fibonacci es calcula obteniendo la suma de los dos terminos anteriores, es decir
fibonacci(n)=fibonacci(n-1) + fibonacci(n-2)
fibonacci(0)=1;
fibonacci(1)=1;

public static void main(String[] args) {
        int num;

        Scanner sc = new Scanner(System.in);
        System.out.println("Dime un numero");
        num = sc.nextInt();
        System.out.println(fib(num));
    }

    static int fib(int num) {
        if (num == 0) {
            return 0;
        } else if (num == 1) {
            return 1;
        } else {
            return fib(num-1)+fib(num-2);
        }
    }
